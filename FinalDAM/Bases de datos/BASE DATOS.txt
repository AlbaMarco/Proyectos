// > many-to-one; < one-to-many; - one-to-one; <> many-to-many

Table USERS as US{
  USER varchar [pk]
  PASS varchar [not null]
  ULT_VISITA varchar
  LEVELU int [ref: > N.NIVEL]
  LEVELA int [ref: > A.NIVEL]
  Indexes {
    (LEVELU, LEVELA)
  }
}

Table NIVELUSERS as N {
  NIVEL int [pk]
  NOMBRE varchar
}

Table NIVELADMIN as A {
  NIVEL int [pk]
  NOMBRE varchar
}

Table LOGADMIN as LOGA {
  USER varchar [pk, ref: > US.USER]
  NIVEL int [ref: > A.NIVEL] 
  ACCIONES varchar [not null]
  FECHA_ACCIONES DATE [not null
}

Table FAVORITOS_USERS {
  USER varchar [pk, ref: > US.USER]
  PKM_FAV varchar [null] // json
  TIPOPKM_FAV varchar [null] // json
  ATAQUES_FAV varchar [null] // json
  HABILIDADES_FAV varchar [null] // json
  OBJETOS_FAV varchar [null] // json
  BAYAS_FAV varchar [null] // json
  // añadiendo los diferetens favoritos que haya.
}

Table EQUIPOSPKM as team {
  ID_USER varchar [pk, ref: > US.USER]
  ID_EQUIPO int [not null, pk]
  ID_POKEMONS [NULL] // json con todos los pkms
  ENFRENTAMIENTOS_JUGADOS int [null]
  ENFRENTAMIENTOS_GANADOS int [null]
  ENFRENTAMIENTOS_PERDIDOS int [null]
}

Table ENFRENTAMIENTOS {
  ID_ENFRENTAMIENTO int [pk, increment]
  ID_TEAM varchar [pk, ref: > EQUIPOSPKM.ID_EQUIPO]
  PKM_DERROTADOS int [not null]
  PKM_ABATIDOS int [not null]
}

Table EQUIPOS_ALEATORIOS {
	ID_EQUIPO_ALEATORIO int [pk]
	ID_POKEMONS varchar [NULL] // json con todos los pkms
	ENFRENTAMIENTOS_JUGADOS int [null]
	ENFRENTAMIENTOS_GANADOS int [null]
	ENFRENTAMIENTOS_PERDIDOS int [null]
}
