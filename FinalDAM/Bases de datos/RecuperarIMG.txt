na vez que hayas insertado la imagen en la base de datos como un BLOB y deseas recuperarla en C# como un objeto de tipo Image, puedes hacer lo siguiente:

using System;
using System.Data.SqlClient;
using System.Drawing;
using System.IO;

class Program
{
    static void Main()
    {
        // Conexión a la base de datos
        string connectionString = "Data Source=NombreServidor;Initial Catalog=NombreBaseDatos;User ID=NombreUsuario;Password=Contraseña";
        using (SqlConnection connection = new SqlConnection(connectionString))
        {
            connection.Open();

            // Consulta SQL para recuperar la imagen
            string sql = "SELECT imagen FROM personajes WHERE nombre_castellano = 'Juan Pérez'"; // Reemplaza con tu propia consulta

            using (SqlCommand command = new SqlCommand(sql, connection))
            {
                using (SqlDataReader reader = command.ExecuteReader())
                {
                    if (reader.Read())
                    {
                        // Leer la imagen como arreglo de bytes
                        byte[] imagenBytes = (byte[])reader["imagen"];

                        // Decodificar la imagen desde base64
                        string imagenCodificada = Convert.ToBase64String(imagenBytes);
                        byte[] imagenDecodificada = Convert.FromBase64String(imagenCodificada);

                        // Crear un MemoryStream para cargar la imagen en memoria
                        using (MemoryStream ms = new MemoryStream(imagenDecodificada))
                        {
                            // Crear un objeto Image a partir del MemoryStream
                            Image imagen = Image.FromStream(ms);

                            // Usar el objeto Image como necesites
                            imagen.Save("imagen.jpg", System.Drawing.Imaging.ImageFormat.Jpeg); // Ejemplo: guardar la imagen en un archivo
                            //imagen.Show(); // Ejemplo: mostrar la imagen en una ventana
                        }
                    }
                }
            }
        }
    }
}


En este ejemplo, se utiliza la clase SqlConnection para establecer una conexión con la base de datos y la clase SqlCommand para ejecutar la consulta SQL. Se lee la imagen como un arreglo de bytes desde la columna BLOB utilizando SqlDataReader, se decodifica desde base64 utilizando Convert.FromBase64String y se carga en un MemoryStream. Luego, se crea un objeto Image a partir del MemoryStream utilizando Image.FromStream, y se puede usar el objeto Image como necesites, como guardarlo en un archivo o mostrarlo en una ventana, según tus necesidades.
